cd "${projectBaseDir}"
set mvn=${mavenHome}\bin\mvn.cmd
set vtp=${versionType}
IF "%vtp%"=="MAJOR" (
    %mvn% build-helper:parse-version versions:set -DnewVersion=${parsedVersion.nextMajorVersion}.0.0 versions:commit
) ELSE IF "%vtp%"=="MINOR" (
    %mvn% build-helper:parse-version versions:set -DnewVersion=${parsedVersion.majorVersion}.${parsedVersion.nextMinorVersion}.0 versions:commit
) ELSE (
    %mvn% build-helper:parse-version versions:set -DnewVersion=${parsedVersion.majorVersion}.${parsedVersion.minorVersion}.${parsedVersion.incrementalVersion} versions:commit
)
